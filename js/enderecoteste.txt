
                // Adiciona o evento de clique para atualizar o endereço selecionado
                $(".click-endereco").on('click', function (e) {
                    e.preventDefault();
                    const enderecoId = $(this).data('entrega-id');
                    const endEntregaRua = $(this).data('entrega-rua');
                    const endEntregaNumero = $(this).data('entrega-numero');
                    const endEntregaComplemento = $(this).data('entrega-complemento');
                    const endEntregaBairro = $(this).data('entrega-bairro');
                    const endEntregaCidade = $(this).data('entrega-cidade');
                    const endEntregaEstado = $(this).data('entrega-estado');
                    const endEntregaCep = $(this).data('entrega-cep');

        
                    // Atualiza o HTML com o endereço selecionado
                    $('#enderecoRua').html(endEntregaRua);
                    $('#enderecoNumero').html(endEntregaNumero);
                    $('#enderecoComplemento').html(endEntregaComplemento);
                    $('#enderecoBairro').html(endEntregaBairro);
                    $('#enderecoCidade').html(endEntregaCidade);
                    $('#enderecoEstado').html(endEntregaEstado);
                    $('#enderecoCep').html(endEntregaCep);
        
                    // Atualiza o localStorage com o ID do endereço selecionado
                    localStorage.setItem('enderecoEntregaId', enderecoId);
        
                    // Fechar o dialog ou outra ação necessária após seleção do endereço
                    app.popup.close();

                    selecionarEndereco(enderecoId);

                    var toastCenter = app.toast.create({
                        text: `Endereço de entrega alterado`,
                        position: 'center',
                        closeTimeout: 2000,
                    });

                    toastCenter.open();
                });
